219 size_t currStateNum = this->m_CurrTopState.state;
220 LRTableEntry currEntry = this->p_Table.atAction(currStateNum, currTokenName);
221

230 }
231
232 errorNum++;
233
234 if (currEntry.isEmpty) {

237 }
238
239 // if error recovery is not enabled
240 if (errorRecoveryType == ErrorRecoveryType::ERT_NONE) {

244 throw std::logic_error("Cannot continue further with the parse! Error entry encountered; It looks like this string does not belong to the grammar.");

252 }
253 default: {
254 std::string errMsg = std::format("Cannot continue further with the parse! Error entry encountered; It looks like this string does not belong to the grammar.\nCurrent stack: {}\n Current input: {}", toString(this->m_Stack), src);

263 // if the error was resolved, continue to the next iteration

268 inline void LRParser<GrammarT, LexicalAnalyzerT, SymbolT, StateT, ParsingTableT, FSMTableT, InputT>::_reduce(size_t prodNumber)
269 {
270 const std::string_view src = this->get_source_code();
271
272 // get the production
273 const auto& production = this->p_Table.grammar.at(prodNumber);
274
275 StackElementType newState = StackElementType{};
276
277 // execute the action, if any
278 if (auto action = static_cast<void(*)(StackType&, StackElementType&)>(production.postfixAction); action != nullptr)
279 action(this->m_Stack, newState);
280
281 // determine the length of the body of the production
282 const size_t prodBodyLength = production.prodBody.size();
283
284 // pop prodBodyLength elements from the top of the stack and get the next entry
285 this->_pop_states(prodBodyLength);
286 size_t stateNum = this->m_CurrTopState.state;
287 LRTableEntry currEntry = this->p_Table.atGoto(stateNum, production.prodHead.as.nonTerminal);
288 newState.state = currEntry.number;
289
290 // Note: errors are never detected when consulting the GOTO table
291 // this here is just a precaution for possible (probably logic) bugs
292 if (currEntry.type != LRTableEntryType::TET_GOTO) {
293 std::string msg{ std::format("Incorrect entry type! Expected type `GOTO` within function reduce after accessing the GOTO table.\nCurrent stack: {}\n Current input: {}", toString(this->m_Stack), src) };
294 this->p_Logger.log(LoggerInfo::ERR_INVALID_VAL, msg);
295
296 throw std::logic_error("Incorrect entry type! Expected type `GOTO` within function reduce after accessing the GOTO table.");
297 }
298
299 // if the current entry is not an error
300 this->_push_state(newState);
301 }

305 inline bool LRParser<GrammarT, LexicalAnalyzerT, SymbolT, StateT, ParsingTableT, FSMTableT, InputT>::_take_parsing_action(auto& result)
306 {
307 using ParserResultType = decltype(result);
308
309 size_t currStateNum = this->m_CurrTopState.state;
310 TerminalType currTokenName = this->m_CurrInputToken.name;
311 LRTableEntry currEntry = this->p_Table.atAction(currStateNum, currTokenName);
312 const std::string_view src = this->get_source_code();
313
314 switch (currEntry.type)
315 {
316 case LRTableEntryType::TET_ACTION_SHIFT: {
317 StateT s = StateT{ currEntry.number };
318 s.token = this->m_CurrInputToken;
319 this->_push_state(s);
320 this->m_CurrInputToken = this->get_next_token();
321 break;
322 }
323
324 case LRTableEntryType::TET_ACTION_REDUCE:
325 _reduce(currEntry.number);
326 break;
327
328 case LRTableEntryType::TET_ACCEPT: {
329 // get the production
330 const auto& production = this->p_Table.grammar.at(0);
331
332 StackElementType newState = StackElementType{};
333
334 // execute the action, if any
335 if (auto action = static_cast<void(*)(StackType&, StackElementType&, ParserResultType&)>(production.postfixAction); action != nullptr)
336 action(this->m_Stack, newState, result);
337 else
338 std::cout << "ACCEPTED!";
339
340 return true;
341 }
342
343 default: // TODO: ENHANCE THIS
344 this->p_Logger.log(LoggerInfo::FATAL_ERROR, std::format("[Unreachable] Invalid entry type `{}` on switch statement!\nCurrent stack: {}\n Current input: {}", toString(currEntry.type), toString(this->m_Stack), src));
345 std::string srcLoc = this->p_Logger.getCurrSourceLocation();
346 assert(std::format("Source code location:\n{}", srcLoc).data());
347 std::abort();
348 }
349
350 return false;
351 }
