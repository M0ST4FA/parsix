372 template<typename ItemT>
373 inline bool ItemSet<ItemT>::contains(const ItemT& item) const
374 {
375 auto begin = this->m_Set.begin();
376 auto end = this->m_Set.end();
377
378 auto predicate = [this, &item](const ItemT& i) {
379 // 'item' is considered found if we found an 'i' such that:
380 // 1. the first components are the same.
381 // 2. the second compoenent of 'item' is a subset of its 'i's counterpart.
382
383 // check for equality of the first components
384 if (item.dotPos != i.dotPos && item.production != i.production)
385 return false;
386
387 // check if item.lookaheads is a subset of i.lookaheads
388 // i.e. if i.lookaheads includes item.lookaheads
389
390 const auto ibegin = i.lookaheads.begin();
391 const auto iend = i.lookaheads.end();
392 const auto itembegin = item.lookaheads.begin();
393 const auto itemend = item.lookaheads.end();
394
395 return std::includes(ibegin, iend, itembegin, itemend);
396 };
397
398 return (std::find_if(begin, end, predicate) != end);
399 }

488 ItemSet<ItemT> ItemSet<ItemT>::CLOSURE(ProdVecType*const grammarPtr)
489 {
490 // check if the closure is already calculated
491 if (this->m_Closure.size() > 0) {
492 this->p_Logger.logDebug("CLOSURE set for this item set has already been calculated!");
493 return this->m_Closure;
494 }
495
496 ProdVecType& grammar = *grammarPtr;
497 // check if the set is not empty (its closure would therefore be empty)
498 if (this->m_Set.size() == 0) {
499 this->p_Logger.log(LoggerInfo::WARNING, "Item set is emtpy. Returning empty CLOSURE!");
500 return *this;
501 }
502
503 // keep track of alternative productions of symbols
504 std::vector<size_t> alternativeProduction[(size_t)VariableType::NT_COUNT]{};
505 for (size_t prodIndex = 0; const ProductionType & prod : grammar.getVector()) {
506 size_t varIndex = (size_t)prod.prodHead.as.nonTerminal;
507 alternativeProduction[varIndex].push_back(prodIndex);
508 prodIndex++;
509 }
510
511 this->p_Logger.logDebug("\nCALCULATING CLOSURE SET:\n");
512 this->p_Logger.logDebug(std::format("Items set:\n {}", (std::string)*this));
513
514 // initialize the closure to the item set
515 this->m_Closure = ItemSet{this->m_Set};
516
517 // check whether the items are LR(0) or they have lookaheads
518 bool isLR0 = this->m_Closure.at(0).lookaheads.size() == 0;
519 std::vector<std::set<SymbolType>> first{};
520 if (not isLR0)
521 grammar.calculateFIRST(), first = grammar.getFIRST();
522
523 this->p_Logger.logDebug(std::format("The items in this set are LR({})", isLR0 ? "0" : "1"));
524
525 // for every item of the closure, calculate the closure of that item
526 for (size_t i = 0; i < this->m_Closure.size(); i++) {
527 const ItemT item = this->m_Closure.at(i);
528 this->p_Logger.logDebug(std::format("Current item being scanned for CLOSURE:{}",item.toString()));
529 const ProductionType& itemProd = item.production;
530
531 // if the dot is at the end of the production
532 if (itemProd.size() == item.dotPos)
533 continue; // there is nothing more to add
534
535 // get the symbol before the dot; note that it is guaranteed to be a symbol
536 // BUG: item.getActualDotPosition()
537 const SymbolType& symbolAfterDot = itemProd.at(item.getActualDotPosition()).as.gramSymbol;
538 this->p_Logger.logDebug(std::format("Symbol after dot is {}", symbolAfterDot.toString()));
539
540 // check if this symbol is a non-terminal
541 if (symbolAfterDot.isTerminal) {
542 // if it is a terminal, the closure of the item is itself and the itemisalrea there in the closure (since we are checking it now)
543 this->p_Logger.logDebug("No other item has been added to CLOSURE");
544 continue;
545 }
546
547 // if we are here, the symbol is a non-terminal
548 // get all the productions for that non-terminal (all of its alternatives)
549 ProdVecType symbolProductions {};
550 const std::vector<size_t>& prods = alternativeProduction[(size_t)symbolAfterDot.as.nonTerminal];
551 for (size_t prodIndex : prods)
552 symbolProductions.pushProduction(grammar.at(prodIndex));
553
554 this->p_Logger.logDebug(std::format("The alternatives of non-terminal {} are: {}",symbolAfterDot.toString(), symbolProductions.toString()));
555
556 // add the new items to the closure
557 if (isLR0)
558 _add_to_closure_no_lookaheads(symbolProductions);
559 else // it is LR(1)
560 {
561 auto predIsGramSymbol = [](const auto& stackElement) {
562 return stackElement.type == ProdElementType::PET_GRAM_SYMBOL;
563 };
564
565 // calculate the lookaheads
566 size_t prodSize = itemProd.prodBody.size();
567 size_t startIndex = item.getActualDotPosition() + 1;
568 SymbolStringType symString{};
569
570 // the following loop corresponds to: std::copy_if(begin + startIndex, end, symString.symbols, predIsGramSymbol);
571 const auto& prodBody = itemProd.prodBody;
572 for (size_t i = startIndex; i < prodBody.size(); i++) {
573 const auto& se = prodBody.at(i);
574
575 // if `se` is a grammar symbol, conncatenate its corresponding grammar symbol with `symString`
576 if (predIsGramSymbol(se))
577 symString.push_back(se.as.gramSymbol);
578
579 }
580
581 this->p_Logger.logDebug(std::format("For item {} => Initial string after the grammar symbol: {}", item.toString(), symString.toString()));
582
583 // go through each lookahead; each lookahead corresponds to a distinct item
584 for (const SymbolType& lookahead : item.lookaheads) {
585 // the string for the current item
586 SymbolStringType tempSymString = symString;
587 tempSymString.push_back(lookahead);
588
589 // the lookaheads of the current item
590
591 tempSymString.calculateFIRST(first);
592 LookAheadSet lookaheads{tempSymString.getFIRST()};
593
594 this->p_Logger.logDebug(std::format("The lookaheads for the current item is: {}", m0st4fa::toString(lookaheads)));
595
596 // add the item to the CLOSURE set
597 _add_to_closure_lookaheads(symbolProductions, lookaheads);
598 }
599
600 }
601 }
602
603 ItemSet res{ this->m_Closure, true };
604 return res;
