165 ParserResultT LLParser<GrammarT, LexicalAnalyzerT, SymbolT, ParsingTableT, FSMTableT, InputT>::parse(ErrorRecoveryType errRecoveryType)
166 {
167 using StackType = std::vector<StackElementType>;
168
169 ParserResultT res;
170 LoggerInfo info{ LoggerInfo::INFO };
171
172 // Initalize the algorithm, such that the parser is in the initial configuration
173 m_Stack.push_back({ .type = ProdElementType::PET_GRAM_SYMBOL, .as = {.gramSymbol = this->get_start_symbol() } });
174
175 this->m_CurrInputToken = this->getLexicalAnalyzer().get_next_token();
176
184 while (-not m_Stack.empty()) {
185 // get the current symbol on top of the stack, pop it and get the next input token
186 m_CurrTopElement = m_Stack.back();
187 m_Stack.pop_back();
188
189 // switch on the type of the top symbol
190
191 switch (m_CurrTopElement.type) {
192 case ProdElementType::PET_GRAM_SYMBOL:
193 parse_grammar_symbol(errRecoveryType);
194 continue;
195
200 case ProdElementType::PET_SYNTH_RECORD: {
201 // extract the record
202 SynthesizedType topRecord = m_CurrTopElement.as.synRecord;
203 auto action = static_cast<void(*)(StackType&, SynDataType&)>(topRecord.action);
204
205 // execute the action if any
206 if (action)
207 action(this->m_Stack, topRecord.data);
208 continue;
209 }
210
211 case ProdElementType::PET_ACTION_RECORD: {
212 // extract the record
213 ActionType topRecord = m_CurrTopElement.as.actRecord;
214 auto action = static_cast<void(*)(StackType&, ActDataType&)>(topRecord.action);
215
216 // execute the action if any
217 if (action)
218 action(this->m_Stack, topRecord.data);
219
220 continue;
221 }
222
223 default:
224 break;
225 }
226 }
227
228 reset_parser_state();
229 return res;
230 }

236 void LLParser<GrammarT, LexicalAnalyzerT, SymbolT, ParsingTableT, FSMTableT, InputT>::parse_grammar_symbol(ErrorRecoveryType errRecoveryType) {
237 using StackElement = StackElementType;
238
239 LoggerInfo info{ .level = LOG_LEVEL::LL_INFO, .info {.noVal = 0} };
240 const SymbolT topSymbol = m_CurrTopElement.as.gramSymbol;
241
242 // if the symbol at the top of the stack is a terminal symbol
243 if (topSymbol.isTerminal) {
244 // epsilon matches with nothing
245 if (topSymbol == TokenType::EPSILON)
246 return;
247
248 // match it explicitly
249 bool matched = (topSymbol == m_CurrInputToken);
250
251 this->p_Logger.logDebug(std::format("Stack size before: {}", m_Stack.size() + 1));
252 this->p_Logger.log(info, std::format("Matched {:s} with {:s}: {:s}", (std::string)topSymbol, (std::string)m_CurrInputToken, matched ? "true" : "false"));
253
254 // get the next input token
255 m_CurrInputToken = this->getLexicalAnalyzer().get_next_token();
256
257 // if the symbol at the top of the stack is not a terminal symbol and the input token is not matched,
258 if (-not matched)
259 error_recovery(errRecoveryType);
260 }
261 // if the symbol is a non-terminal symbol
262 else {
263 // get the production record for the current symbol and input
264 const LLTableEntry tableEntry = this->p_Table[EXTRACT_VARIABLE(this->m_CurrTopElement)][(size_t)m_CurrInputToken.name];
265
266 // if the table entry is an error
267 if (tableEntry.isError) {
268 // TODO: do more robust logic based on the boolean returned from the function
269 error_recovery(errRecoveryType);
270 return; // assuming we have syncronized and are ready to continue parsing
271 }
272
273 // if the table entry is not an error
274
275 // Caution: this is a reference
276 const auto& prod = this->m_ProdRecords[tableEntry.prodIndex];
277 const auto& prodBody = prod.prodBody;
278
279 // check that the production body is not empty
280 check_prod_body(prod);
281
282 // push the body of the production on top of the stack
283 for (auto it = prodBody.rbegin(); it != prodBody.rend(); ++it) {
284 StackElement se = *it;
285 m_Stack.push_back(se);
286 };
287
288 this->p_Logger.logDebug(std::format("Stack size before: {}", m_Stack.size() + 1));
289 this->p_Logger.log(info, std::format("Expanded {:s} with {:s}: {:s}", (std::string)topSymbol, (std::string)m_CurrInputToken, (std::string)prod));
290 }
291
292 return;
293 }

299 bool LLParser<GrammarT, LexicalAnalyzerT, SymbolT, ParsingTableT, FSMTableT, InputT>::panic_mode()
300 {
301 using StackType = StackType<StackElementType>;
302
303 // get top stack element
304 auto currInputToken = this->m_CurrInputToken;
305 const SymbolT topSymbol = m_CurrTopElement.as.gramSymbol;
306
307 // set up the logger state
308 LoggerInfo info{ .level = LOG_LEVEL::LL_INFO, .info {.noVal = 0} };
309 LoggerInfo errInfo{ .level = LOG_LEVEL::LL_ERROR, .info {.errorType = ERROR_TYPE::ET_UNEXCPECTED_TOKEN } };
310
311 this->p_Logger.log(errInfo, std::format(
312 "({}, {}) Didn't excpect token {:s}",
313 this->getLexicalAnalyzer().getLine(),
314 this->getLexicalAnalyzer().getCol(),
315 (std::string)this->m_CurrInputToken
316 ));
317
318 // loop until the input and the stack are synchronized
319 while (true) {
320 // if the top symbol is a terminal
321 if (topSymbol.isTerminal) {
322 this->p_Logger.log(info, std::format(
323 "Added lexeme {:s} to the input stream.", currInputToken.attribute)
324 );
325
326 // pop the token of the stack and return to the parser
327 m_Stack.pop_back();
328 return true;
329 }
330 // if the top symbol is a non-terminal
331 else {
332 // peak to see the next token
333 currInputToken = this->getLexicalAnalyzer().peak();
334
335 // check if it can be used to sync with the parser
336 bool synced = this->panic_mode_try_sync_variable(currInputToken);
337
338 if (synced)
339 break;
340
341 // if we've reached the end of the input and could not sync with this token
342 if (m_CurrInputToken == TokenType{}) {
343 // pop this element since there is no way to syncronize using it
344 m_Stack.pop_back();
345
346 // if the stack is empty, return to the caller
347 if (m_Stack.empty()) {
348 this->p_Logger.log(info, std::format("[ERROR_RECOVERY] ({}, {}) Failed to syncronize: current input: {:s}",
349 this->getLexicalAnalyzer().getLine(),
350 this->getLexicalAnalyzer().getCol(),
351 (std::string)currInputToken
352 ));
353
354 return false;
355 };
356 };
357 }
358 }
359
360 return true;
361 }

372 bool LLParser<GrammarT, LexicalAnalyzerT, SymbolT, ParsingTableT, FSMTableT, InputT>::panic_mode_try_sync_variable(TokenType& currInputToken) {
373 using StackType = StackType<StackElementType>;
374
375 LoggerInfo info{ .level = LOG_LEVEL::LL_INFO, .info {.noVal = 0} };
376
377 // Check for whether the non-terminal has an epsilon production and use it to reduce that non-terminal.
378 LLTableEntry tableEntry = this->p_Table[EXTRACT_VARIABLE(this->m_CurrTopElement)][(size_t)TokenType::EPSILON.name];
379
380 // if it is not an err, there is an epsilon production
381 if (-not tableEntry.isError) {
382 // expand with this production and assume we are in sync
383
384 // Caution: this is a reference
385 const auto& prod = this->m_ProdRecords[tableEntry.prodIndex];
386 const auto& prodBody = prod.prodBody;
387
388 // check that the production body is not empty
389 check_prod_body(prod);
390
391 // push the body of the production on top of the stack
392 for (auto it = prodBody.rbegin(); it != prodBody.rend(); ++it) {
393 StackElementType se = *it;
394 m_Stack.push_back(se);
395 };
396
397 this->p_Logger.log(info, std::format("[ERROR_RECOVERY] Expanded {:s} with {:s}: {:s}",
398 toString(m_CurrTopElement.as.gramSymbol.as.nonTerminal),
399 m_CurrInputToken.toString(),
400 prod.toString()));
401
402 print_sync_msg(this->getLexicalAnalyzer().getPosition());
403
404 return true;
405 }
406
407 // get the production record for the current symbol and input
408 tableEntry = this->p_Table[EXTRACT_VARIABLE(this->m_CurrTopElement)][(size_t)currInputToken.name];
409
419 if (tableEntry.isError) {
420 // if the entry has an action
421 if (tableEntry.action) {
422 auto action = static_cast<bool (*)(StackType, StackElementType, TokenType)>(tableEntry.action);
423
424 // if the action results in a syncronization
425 if (action(m_Stack, m_CurrTopElement, currInputToken)) {
426 m_CurrInputToken = this->getLexicalAnalyzer().get_next_token();
427
428 print_sync_msg(this->getLexicalAnalyzer().getPosition());
429
430 return true;
431 }
432 }
433
434 // if the entry has no action or the action has failed to syncronize
435
436 // get the next input and try again to syncronize
437 m_CurrInputToken = this->getLexicalAnalyzer().get_next_token();
438 return false;
439 }
440
441 //---------------------------------------------Syncronized-------------------------------------------
446 m_CurrInputToken = this->getLexicalAnalyzer().get_next_token();
447 print_sync_msg(this->getLexicalAnalyzer().getPosition());
448
454 return true;
455 }
